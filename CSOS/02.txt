==== 0 - 10 の偶数の和 ====
MAIN  START
  LD  GR1,=11
  LD  GR7,=0
LOOP  SUBA  GR1,=1
  JZE EXIT
  LD  GR2,GR1
  AND GR2,=#0001
  JNZ LOOP
  ADDA  GR7,GR1
  JUMP LOOP
EXIT  ST  GR7,#0050
  END

==== 12 / 4 ====
MAIN  START
  LD GR1,=12
  SRA GR1,2
  ST GR1,#0050
  END

==== fib(20) ====
MAIN  START
  LD GR7,=1
  LD GR1,GR7
  LD GR2,GR7
  ST GR1,#0050
  ST GR1,#0051
FIB ADDA GR2,GR1
  ST GR2,#0051,GR7
  LD GR6,GR2
  SUBA GR6,GR1
  LD GR1,GR6
  ADDA GR7,=1
  CPL GR7,=19
  JNZ FIB
  END
